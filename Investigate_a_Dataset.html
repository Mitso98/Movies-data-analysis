# adjust relase date so it becomes 4 digits rather than 2
year_lst = list(popular.index)

for i,year in enumerate(year_lst):
    if int(year) >= 1 and int(year) <= 15:
        year_lst[i] = '20' + year
    elif year == '00':
        year_lst[i] = 2000    
    else:
        year_lst[i] = '19' + year
